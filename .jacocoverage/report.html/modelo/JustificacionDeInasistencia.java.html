package modelo;

import excepciones.*;
import dao.*;
import java.sql.Time;
import org.joda.time.DateTime;

public class JustificacionDeInasistencia {
    
    private Time horaActual;
    private Inasistencia inasistencia;
    private DateTime fechaActual;
    private IServicioInasistencia servicio;
    
    public JustificacionDeInasistencia(Inasistencia inasistencia, 
            Time horaActual, DateTime fechaActual) {
        
        this.inasistencia = inasistencia;
        this.horaActual = horaActual;
        this.fechaActual = fechaActual;
    }
   
    public void Justificar(String descripcion) throws HorasDeInasistenciaExcedida, 
            InasistenciaJustificadaException {
        
        if (inasistencia.estaJustificada()){
            throw new InasistenciaJustificadaException();
        } else if (tiene24HorasGenerada()) {
            throw new HorasDeInasistenciaExcedida();
        }
        inasistencia.CrearJustificacion(descripcion);
        servicio.guardar(inasistencia);
    }
    
    private boolean tiene24HorasGenerada() {
        return esOtroDia() && 
                horaActual.getTime() >= inasistencia.getHoraDeGeneracion().getTime();
    }
    
    private boolean esOtroDia() {
        return inasistencia.getDia().isBefore(fechaActual);
    }
    
    public void setServicio(IServicioInasistencia servI) {
        this.servicio = servI;
    }
}
